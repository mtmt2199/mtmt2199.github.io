<!DOCTYPE html>
<html lang="ja">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>HTMLファイル読み込み</title>
</head>
<body>
    <h1>別ファイルの内容を表示します</h1>
    <div id="content">
        </div>

    <button onclick="loadAnotherHtml()">別のHTMLを読み込む</button>

    <script>
        function loadAnotherHtml() {
            const contentDiv = document.getElementById('content');
            const fileName = 'another.html'; // 同一階層にあるHTMLファイル名

            fetch(fileName)
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }
                    return response.text();
                })
                .then(htmlContent => {
                    contentDiv.innerHTML = htmlContent;
                })
                .catch(error => {
                    console.error('HTMLファイルの読み込み中にエラーが発生しました:', error);
                    contentDiv.innerHTML = '<p>ファイルの読み込みに失敗しました。</p>';
                });
        }
    </script>
</body>
</html>
